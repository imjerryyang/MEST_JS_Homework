 *  question, please make sure you do your research on this).
 *
 */

email validation can be done using regular expressions. 
function validateEmail(email) {
	var x = /^(([^<>()\[\]\\.,;:\s@"]+)|(".+"))@((\[[0-9]{1,3}\.[0-9]{1,3\.[0-9]{1,3}\.[0-9[{1,3}])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
	return x.test(email);
}
